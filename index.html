Q1-Find the Smaller Angle-

        var Minimal_Angle = (h, m) => 
        {
              h=h%12;
              let hour= (h*360)/12 + (m*360)/ (12*60);
              let min = (m*360)/60;
              let angle= Math.abs(hour-min);
              if (angle>180)
              return 360-angle;
              else
              return angle;
        };

Q4-Reverse a Number-

        var Reverse_Number = (N) => 
        {
         let x=0;
         while(N>0) {
           let temp=N%10;
           x=x*10 + temp;
           N=parseInt(N/10);
         }
         return x;
        };
